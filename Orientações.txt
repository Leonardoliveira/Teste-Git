comando do git

--para inicializar um repositório 

git init

--para ver a versão do git
git --version

--para ver o status do repositorio
git status

--adicionar um arquivo do git
git add <nome do arquivo>

--adicionando todos os arquivos da pasta atual
git add .

--Fazendo o comentário
git commit -m "mensagem a descrever"

--fazendo login
git config --local user.name "Seu nome aqui"
git config --local user.email "seu@email.aqui"

--vendo um relatório do git
git log

--Fazendo a configuração do nome
git config --local user.name

--Fazendo a configuração do email
git config --local user.email

--visualizando as informações em uma linha
git log --oneline

--visualizando mais detalhado
git log -p

--criando um repositório para apenas as alterações
git init --bare

--listar todos os repositorio remotos
git remote

--adicionar um repositorio remoto
git remote add [nome do repositorio] [caminho do repositorio]

--clona o repositorio
git clone

-- enviar os dados para o repositorio
git push [nome do repositorio] master

--baixa os arquivos do repositorio
git pull

--vendo as branch
git branch

--criando uma branch
git branch [nome]

--mudando para a branch 
git checkout [nome da branch]

--criando uma branch e habilitar
git checkout -b [nome da branch]

--unificando a branch
git merge [nome da branch]

--atualizando a branch master
git rebase [ñome da branch]
